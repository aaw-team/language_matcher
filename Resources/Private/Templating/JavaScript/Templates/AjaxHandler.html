httpRequest = new XMLHttpRequest(); httpRequest.onreadystatechange = function(e){ var request = e.target; if (request.readyState === XMLHttpRequest.DONE) { if (request.status === 200) { var response = JSON.parse(request.responseText); if (typeof response === 'object' && response.showHint) { var el = document.createElement('div'); // el.innerHTML = 'Go to ' + response.title + ''; el.innerHTML = response.html; document.body.appendChild(el); } } } }; httpRequest.open('GET', '{ajaxUri}', true); httpRequest.send();
